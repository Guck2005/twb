## Schéma de base

Considérons les tables suivantes :


CREATE TABLE Departement (
    id_dept NUMBER PRIMARY KEY,
    nom_dept VARCHAR2(50)
);

CREATE TABLE Employe (
    id_emp NUMBER PRIMARY KEY,
    nom VARCHAR2(50),
    salaire NUMBER(10,2),
    date_embauche DATE,
    id_dept NUMBER,
    CONSTRAINT fk_dept FOREIGN KEY (id_dept)
        REFERENCES Departement(id_dept)
);

CREATE TABLE Historique_Salaire (
    id_hist NUMBER PRIMARY KEY,
    id_emp NUMBER,
    ancien_salaire NUMBER(10,2),
    nouveau_salaire NUMBER(10,2),
    date_modification DATE
);

Partie 1 : Procédures
Exercice 1 : Augmentation de salaire
CREATE OR REPLACE PROCEDURE augmenter_salaire (
    p_id_emp IN Employe.id_emp%TYPE,
    p_pourcentage IN NUMBER
)
IS
BEGIN
    UPDATE Employe
    SET salaire = salaire + (salaire * p_pourcentage / 100)
    WHERE id_emp = p_id_emp;

    COMMIT;
END;
/
Exercice 2 : Prime départementale
CREATE OR REPLACE PROCEDURE prime_departement (
    p_id_dept IN Departement.id_dept%TYPE
)
IS
BEGIN
    UPDATE Employe
    SET salaire = salaire * 1.05
    WHERE id_dept = p_id_dept;

    COMMIT;
END;
/
Exercice 3 : Embauche d'un employé
CREATE OR REPLACE PROCEDURE embaucher_employe (
    p_id_emp IN NUMBER,
    p_nom IN VARCHAR2,
    p_salaire IN NUMBER,
    p_id_dept IN NUMBER
)
IS
    v_count NUMBER;

    dept_inexistant EXCEPTION;
BEGIN
    SELECT COUNT(*)
    INTO v_count
    FROM Departement
    WHERE id_dept = p_id_dept;

    IF v_count = 0 THEN
        RAISE dept_inexistant;
    END IF;

    INSERT INTO Employe(
        id_emp,
        nom,
        salaire,
        date_embauche,
        id_dept
    )
    VALUES(
        p_id_emp,
        p_nom,
        p_salaire,
        SYSDATE,
        p_id_dept
    );

    COMMIT;

EXCEPTION
    WHEN dept_inexistant THEN
        RAISE_APPLICATION_ERROR(
            -20001,
            'Département inexistant'
        );
END;
/
Exercice 4 : Rapport départemental
CREATE OR REPLACE PROCEDURE rapport_departement (
    p_id_dept IN NUMBER
)
IS
    v_nb NUMBER;
    v_moyenne NUMBER;
    v_max NUMBER;
BEGIN
    SELECT COUNT(*),
           AVG(salaire),
           MAX(salaire)
    INTO v_nb,
         v_moyenne,
         v_max
    FROM Employe
    WHERE id_dept = p_id_dept;

    DBMS_OUTPUT.PUT_LINE(
        'Nombre employés : ' || v_nb
    );

    DBMS_OUTPUT.PUT_LINE(
        'Salaire moyen : ' || NVL(v_moyenne,0)
    );

    DBMS_OUTPUT.PUT_LINE(
        'Salaire maximum : ' || NVL(v_max,0)
    );
END;
/
Partie 2 : Fonctions
Exercice 5 : Salaire annuel
CREATE OR REPLACE FUNCTION salaire_annuel (
    p_id_emp IN NUMBER
)
RETURN NUMBER
IS
    v_salaire Employe.salaire%TYPE;
BEGIN
    SELECT salaire
    INTO v_salaire
    FROM Employe
    WHERE id_emp = p_id_emp;

    RETURN v_salaire * 12;
END;
/
Exercice 6 : Ancienneté
CREATE OR REPLACE FUNCTION anciennete (
    p_id_emp IN NUMBER
)
RETURN NUMBER
IS
    v_date DATE;
BEGIN
    SELECT date_embauche
    INTO v_date
    FROM Employe
    WHERE id_emp = p_id_emp;

    RETURN TRUNC(MONTHS_BETWEEN(
        SYSDATE,
        v_date
    ) / 12);
END;
/
Exercice 7 : Classification salariale
CREATE OR REPLACE FUNCTION classification_salariale (
    p_salaire IN NUMBER
)
RETURN VARCHAR2
IS
BEGIN
    IF p_salaire < 1000 THEN
        RETURN 'Junior';

    ELSIF p_salaire <= 3000 THEN
        RETURN 'Confirme';

    ELSE
        RETURN 'Senior';
    END IF;
END;
/
Exercice 8 : Masse salariale
CREATE OR REPLACE FUNCTION masse_salariale (
    p_id_dept IN NUMBER
)
RETURN NUMBER
IS
    v_total NUMBER;
BEGIN
    SELECT NVL(SUM(salaire),0)
    INTO v_total
    FROM Employe
    WHERE id_dept = p_id_dept;

    RETURN v_total;
END;
/
Partie 3 : Triggers
Exercice 9 : Historisation des salaires

Comme id_hist est une clé primaire, on crée d'abord une séquence :

CREATE SEQUENCE seq_hist
START WITH 1
INCREMENT BY 1;

Trigger :

CREATE OR REPLACE TRIGGER trg_historique_salaire
BEFORE UPDATE OF salaire
ON Employe
FOR EACH ROW
BEGIN
    INSERT INTO Historique_Salaire(
        id_hist,
        id_emp,
        ancien_salaire,
        nouveau_salaire,
        date_modification
    )
    VALUES(
        seq_hist.NEXTVAL,
        :OLD.id_emp,
        :OLD.salaire,
        :NEW.salaire,
        SYSDATE
    );
END;
/
Exercice 10 : Contrôle du salaire
CREATE OR REPLACE TRIGGER trg_controle_salaire
BEFORE INSERT OR UPDATE OF salaire
ON Employe
FOR EACH ROW
BEGIN
    IF :NEW.salaire < 0 THEN
        RAISE_APPLICATION_ERROR(
            -20002,
            'Salaire negatif interdit'
        );
    END IF;
END;
/
Exercice 11 : Date d'embauche automatique
CREATE OR REPLACE TRIGGER trg_date_embauche
BEFORE INSERT
ON Employe
FOR EACH ROW
BEGIN
    IF :NEW.date_embauche IS NULL THEN
        :NEW.date_embauche := SYSDATE;
    END IF;
END;
/
Exercice 12 : Limitation des augmentations et suppression protégée
a) Augmentation maximale de 30 %
CREATE OR REPLACE TRIGGER trg_limite_augmentation
BEFORE UPDATE OF salaire
ON Employe
FOR EACH ROW
BEGIN
    IF :NEW.salaire >
       :OLD.salaire * 1.30 THEN

       RAISE_APPLICATION_ERROR(
           -20003,
           'Augmentation superieure a 30% interdite'
       );
    END IF;
END;
/
b) Interdire la suppression si le département contient moins de 3 employés
CREATE OR REPLACE TRIGGER trg_interdire_suppression
BEFORE DELETE
ON Employe
FOR EACH ROW
DECLARE
    v_nb NUMBER;
BEGIN
    SELECT COUNT(*)
    INTO v_nb
    FROM Employe
    WHERE id_dept = :OLD.id_dept;

    IF v_nb < 3 THEN
        RAISE_APPLICATION_ERROR(
            -20004,
            'Suppression interdite : moins de 3 employes dans le departement'
        );
    END IF;
END;
/
Remarque importante pour l'exercice 12(b)

Dans Oracle, ce trigger peut provoquer une erreur de table mutante (mutating table) car il interroge la table Employe pendant une suppression sur cette même table.
